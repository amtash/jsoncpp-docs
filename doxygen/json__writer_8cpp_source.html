78 std::string valueToString(double value) {
79  // Allocate a buffer that is more than large enough to store the 16 digits of
80  // precision requested below.
81  char buffer[32];
82  int len = -1;
83 
84 // Print into the buffer. We need not request the alternative representation
85 // that always has a decimal point because JSON doesn't distingish the
86 // concepts of reals and integers.
87 #if defined(_MSC_VER) && defined(__STDC_SECURE_LIB__) // Use secure version with
88  // visual studio 2005 to
89  // avoid warning.
90 #if defined(WINCE)
91  len = _snprintf(buffer, sizeof(buffer), "%.17g", value);
92 #else
93  len = sprintf_s(buffer, sizeof(buffer), "%.17g", value);
94 #endif
95 #else
96  if (isfinite(value)) {
97  len = snprintf(buffer, sizeof(buffer), "%.17g", value);
98  } else {
99  // IEEE standard states that NaN values will not compare to themselves
100  if (value != value) {
101  len = snprintf(buffer, sizeof(buffer), "null");
102  } else if (value < 0) {
103  len = snprintf(buffer, sizeof(buffer), "-1e+9999");
104  } else {
105  len = snprintf(buffer, sizeof(buffer), "1e+9999");
106  }
107  // For those, we do not need to call fixNumLoc, but it is fast.
108  }
109 #endif
110  assert(len >= 0);
111  fixNumericLocale(buffer, buffer + len);
112  return buffer;
113 }

117 std::string valueToQuotedString(const char* value) {
118  if (value == NULL)
119  return "";
120  // Not sure how to handle unicode...
121  if (strpbrk(value, "\"\\\b\f\n\r\t") == NULL &&
122  !containsControlCharacter(value))
123  return std::string("\"") + value + "\"";
124  // We have to walk value and escape any special characters.
125  // Appending to std::string is not efficient, but this should be rare.
126  // (Note: forward slashes are *not* rare, but I am not escaping them.)
127  std::string::size_type maxsize =
128  strlen(value) * 2 + 3; // allescaped+quotes+NULL
129  std::string result;
130  result.reserve(maxsize); // to avoid lots of mallocs
131  result += "\"";
132  for (const char* c = value; *c != 0; ++c) {
133  switch (*c) {
134  case '\"':
135  result += "\\\"";
136  break;
137  case '\\':
138  result += "\\\\";
139  break;
140  case '\b':
141  result += "\\b";
142  break;
143  case '\f':
144  result += "\\f";
145  break;
146  case '\n':
147  result += "\\n";
148  break;
149  case '\r':
150  result += "\\r";
151  break;
152  case '\t':
153  result += "\\t";
154  break;
155  // case '/':
156  // Even though \/ is considered a legal escape in JSON, a bare
157  // slash is also legal, so I see no reason to escape it.
158  // (I hope I am not misunderstanding something.
159  // blep notes: actually escaping \/ may be useful in javascript to avoid </
160  // sequence.
161  // Should add a flag to allow this compatibility mode and prevent this
162  // sequence from occurring.
163  default:
164  if (isControlCharacter(*c)) {
165  std::ostringstream oss;
166  oss << "\\u" << std::hex << std::uppercase << std::setfill('0')
167  << std::setw(4) << static_cast<int>(*c);
168  result += oss.str();
169  } else {
170  result += *c;
171  }
172  break;
173  }
174  }
175  result += "\"";
176  return result;
177 }

215  break;

218  break;
219  case stringValue:
220  document_ += valueToQuotedString(value.asCString());
221  break;
222  case booleanValue:

224  break;
225  case arrayValue: {
226  document_ += '[';
227  int size = value.size();
228  for (int index = 0; index < size; ++index) {
229  if (index > 0)

364 bool StyledWriter::isMultineArray(const Value& value) {
365  int size = value.size();
366  bool isMultiLine = size * 3 >= rightMargin_;
367  childValues_.clear();
368  for (int index = 0; index < size && !isMultiLine; ++index) {
369  const Value& childValue = value[index];
370  isMultiLine =
371  isMultiLine || ((childValue.isArray() || childValue.isObject()) &&
372  childValue.size() > 0);
373  }
374  if (!isMultiLine) // check if line length > max line length
375  {
376  childValues_.reserve(size);
377  addChildValues_ = true;
378  int lineLength = 4 + (size - 1) * 2; // '[ ' + ', '*n + ' ]'
379  for (int index = 0; index < size; ++index) {
380  if (hasCommentForValue(value[index])) {
381  isMultiLine = true;
382  }
383  writeValue(value[index]);
384  lineLength += int(childValues_[index].length());
385  }
386  addChildValues_ = false;
387  isMultiLine = isMultiLine || lineLength >= rightMargin_;
388  }
389  return isMultiLine;

753  pushValue(valueToQuotedString(value.asCString()));
754  break;
755  case booleanValue:
756  pushValue(valueToString(value.asBool()));
757  break;
758  case arrayValue:
759  writeArrayValue(value);
760  break;
761  case objectValue: {
762  Value::Members members(value.getMemberNames());
763  if (members.empty())
764  pushValue("{}");
765  else {
766  writeWithIndent("{");
767  indent();
768  Value::Members::iterator it = members.begin();
769  for (;;) {
770  std::string const& name = *it;
771  Value const& childValue = value[name];
772  writeCommentBeforeValue(childValue);
773  writeWithIndent(valueToQuotedString(name.c_str()));
774  sout_ << colonSymbol_;
775  writeValue(childValue);
776  if (++it == members.end()) {
777  writeCommentAfterValueOnSameLine(childValue);
778  break;
779  }
780  sout_ << ",";
781  writeCommentAfterValueOnSameLine(childValue);
782  }
783  unindent();
784  writeWithIndent("}");
785  }
786  } break;
787  }
788 }

790 void BuiltStyledStreamWriter::writeArrayValue(Value const& value) {
791  unsigned size = value.size();
792  if (size == 0)
793  pushValue("[]");
794  else {
795  bool isMultiLine = (cs_ == CommentStyle::All) || isMultineArray(value);
796  if (isMultiLine) {
797  writeWithIndent("[");
798  indent();
799  bool hasChildValue = !childValues_.empty();
800  unsigned index = 0;
801  for (;;) {
802  Value const& childValue = value[index];
803  writeCommentBeforeValue(childValue);
804  if (hasChildValue)
805  writeWithIndent(childValues_[index]);
806  else {
807  if (!indented_) writeIndent();
808  indented_ = true;
809  writeValue(childValue);
810  indented_ = false;
811  }
812  if (++index == size) {
813  writeCommentAfterValueOnSameLine(childValue);
814  break;
815  }
816  sout_ << ",";
817  writeCommentAfterValueOnSameLine(childValue);
818  }
819  unindent();
820  writeWithIndent("]");
821  } else // output on a single line
822  {
823  assert(childValues_.size() == size);
824  sout_ << "[";
825  if (!indentation_.empty()) sout_ << " ";
826  for (unsigned index = 0; index < size; ++index) {
827  if (index > 0)
828  sout_ << ", ";
829  sout_ << childValues_[index];
830  }
831  if (!indentation_.empty()) sout_ << " ";
832  sout_ << "]";
833  }
834  }
835 }

837 bool BuiltStyledStreamWriter::isMultineArray(Value const& value) {
838  int size = value.size();
839  bool isMultiLine = size * 3 >= rightMargin_;
840  childValues_.clear();
841  for (int index = 0; index < size && !isMultiLine; ++index) {
842  Value const& childValue = value[index];
843  isMultiLine =
844  isMultiLine || ((childValue.isArray() || childValue.isObject()) &&
845  childValue.size() > 0);
846  }
847  if (!isMultiLine) // check if line length > max line length
848  {
849  childValues_.reserve(size);
850  addChildValues_ = true;
851  int lineLength = 4 + (size - 1) * 2; // '[ ' + ', '*n + ' ]'
852  for (int index = 0; index < size; ++index) {
853  if (hasCommentForValue(value[index])) {
854  isMultiLine = true;
855  }
856  writeValue(value[index]);
857  lineLength += int(childValues_[index].length());
858  }
859  addChildValues_ = false;
860  isMultiLine = isMultiLine || lineLength >= rightMargin_;
861  }
862  return isMultiLine;
863 }
